# worker-deploy.yml
# Two trigger paths:
#   1. workflow_dispatch — manual staging deploy from any branch/SHA
#   2. push to master (paths: proxy/**) — auto-triggers production deploy
#      Production requires approval from the worker-production GitHub Environment.
#
# Owner: Cassian (SRE) — production approval gate lives here.
# Arya is backup reviewer on worker-production environment.

name: Worker Deploy

on:
  workflow_dispatch:
    inputs:
      environment:
        description: 'Target environment'
        required: true
        type: choice
        options:
          - staging
          - production
      ref:
        description: 'Branch or SHA to deploy (default: master)'
        required: false
        default: 'master'
  push:
    branches:
      - master
    paths:
      - 'proxy/**'
      - '.github/workflows/worker-deploy.yml'

jobs:
  resolve-env:
    name: Resolve target environment
    runs-on: ubuntu-latest
    outputs:
      environment: ${{ steps.set.outputs.environment }}
    steps:
      - id: set
        shell: bash
        run: |
          if [ "${{ github.event_name }}" = "push" ]; then
            echo "environment=production" >> $GITHUB_OUTPUT
          else
            echo "environment=${{ inputs.environment }}" >> $GITHUB_OUTPUT
          fi

  deploy:
    name: Deploy to ${{ needs.resolve-env.outputs.environment }}
    needs: resolve-env
    runs-on: ubuntu-latest

    # GitHub Environment gate:
    # - worker-staging: no approval required, auto-deploy
    # - worker-production: requires Cassian approval (Arya as backup)
    environment:
      name: worker-${{ needs.resolve-env.outputs.environment }}

    env:
      TARGET_ENV: ${{ needs.resolve-env.outputs.environment }}

    defaults:
      run:
        working-directory: proxy

    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          ref: ${{ inputs.ref || github.sha }}

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: proxy/package-lock.json

      - name: Install dependencies (lockfile-backed)
        run: npm ci

      - name: Type check
        run: npx tsc --noEmit

      - name: Unit tests
        run: npm test

      - name: Secret leak scan
        shell: bash
        run: |
          echo "Scanning proxy/ for leaked secrets..."
          HITS=$(grep -rEn 'sk-ant-|AIza[0-9A-Za-z_-]{35}|ANTHROPIC_API_KEY|GEMINI_API_KEY' . \
            --include="*.ts" --include="*.js" --include="*.json" --include="*.toml" \
            --exclude-dir=node_modules \
            --exclude="*.test.ts" --exclude="*.spec.ts" \
            || true)
          if [ -n "$HITS" ]; then
            echo ""
            echo "FAIL: Secret leak detected. Aborting deploy."
            echo "$HITS"
            exit 1
          fi
          echo "PASS: No secrets found"

      - name: Deploy Worker
        id: deploy
        run: npx wrangler deploy --env ${{ env.TARGET_ENV }}
        env:
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}

      - name: Smoke tests
        id: smoke
        run: node scripts/smoke.js
        env:
          SMOKE_ENV: ${{ env.TARGET_ENV }}
          PROXY_URL: ${{ env.TARGET_ENV == 'production' && secrets.PROXY_URL_PRODUCTION || secrets.PROXY_URL_STAGING }}
          PROXY_TOKEN: ${{ env.TARGET_ENV == 'production' && secrets.WORKER_CLIENT_TOKEN_PRODUCTION || secrets.WORKER_CLIENT_TOKEN_STAGING }}

      # If smoke tests fail post-deploy, roll back immediately.
      # Rollback targets the last known-good Cloudflare deployment.
      # Alert is always emitted on rollback regardless of other failures.
      - name: Rollback on smoke failure
        if: failure() && steps.deploy.outcome == 'success' && steps.smoke.outcome == 'failure'
        shell: bash
        run: |
          echo "WARNING: Smoke tests failed post-deploy. Initiating rollback..."
          npx wrangler rollback --env ${{ env.TARGET_ENV }}
          echo "Rollback complete. Investigate before re-deploying."
        env:
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
