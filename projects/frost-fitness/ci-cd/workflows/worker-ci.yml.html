# worker-ci.yml
# Runs on every PR that touches the Worker or its workflow config.
# Purpose: fast feedback loop — type safety, tests, secret scan, dry-run.
# Does NOT deploy. Deploy is owned by worker-deploy.yml.
#
# Owner: Cassian (SRE)

name: Worker CI

on:
  pull_request:
    paths:
      - 'proxy/**'
      - '.github/workflows/worker-*.yml'

jobs:
  check:
    name: Check (type · lint · test · scan · dry-run)
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: proxy

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: proxy/package-lock.json

      - name: Install dependencies (lockfile-backed)
        run: npm ci

      - name: Type check
        run: npx tsc --noEmit

      - name: Lint
        run: npm run lint

      - name: Unit tests
        run: npm test

      - name: Secret leak scan
        shell: bash
        run: |
          echo "Scanning proxy/ for leaked secrets..."
          HITS=$(grep -rEn 'sk-ant-|AIza[0-9A-Za-z_-]{35}|ANTHROPIC_API_KEY|GEMINI_API_KEY' . \
            --include="*.ts" --include="*.js" --include="*.json" --include="*.toml" \
            --exclude-dir=node_modules \
            --exclude="*.test.ts" --exclude="*.spec.ts" \
            || true)
          if [ -n "$HITS" ]; then
            echo ""
            echo "FAIL: Secret leak detected. Remove these before merging:"
            echo "$HITS"
            exit 1
          fi
          echo "PASS: No secrets found in proxy/"

      - name: Validate secret scan catches leaks
        shell: bash
        run: |
          echo "Validating scanner with injected fake key..."
          echo 'const fakeKey = "sk-ant-test-cassian-validation-key-abc123";' > /tmp/leak-test.ts
          HITS=$(grep -rEn 'sk-ant-' /tmp/leak-test.ts || true)
          if [ -z "$HITS" ]; then
            echo "FAIL: Scanner did not catch the injected key. Fix the scan pattern."
            exit 1
          fi
          rm /tmp/leak-test.ts
          echo "PASS: Scanner correctly catches injected keys"

      - name: Wrangler dry run (staging config)
        run: npx wrangler deploy --dry-run --env staging
        env:
          CLOUDFLARE_API_TOKEN: ${{ secrets.CLOUDFLARE_API_TOKEN }}
          CLOUDFLARE_ACCOUNT_ID: ${{ secrets.CLOUDFLARE_ACCOUNT_ID }}
