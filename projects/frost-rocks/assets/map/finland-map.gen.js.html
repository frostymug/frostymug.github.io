// Frost Rocks — Finland interim offline map raster generator (D5).
// Field-guide-styled, "simple pixelated" raster. Places drawn at true geographic
// positions using the SAME linear projection Jesse's projectCoord() will use, so
// the handoff bounds are exact.
const fs = require('fs');
const path = require('path');

// ---- BOUNDS (the load-bearing contract with app code) --------------------
// NW = top-left corner, SE = bottom-right corner.
const BOUNDS = { nwLat: 64.0, nwLng: 23.0, seLat: 59.5, seLng: 31.0 };

// Canvas size. Aspect chosen to match real km (cos-lat corrected) so shapes
// aren't distorted. Height/width ~1.18 (see genmap notes).
const W = 1024;
const H = 1208;

// projectCoord — MUST match the app-side function exactly.
// Linear interpolation between corners. lng -> x, lat -> y (y inverted: north=top).
function project(lat, lng) {
  const x = ((lng - BOUNDS.nwLng) / (BOUNDS.seLng - BOUNDS.nwLng)) * W;
  const y = ((BOUNDS.nwLat - lat) / (BOUNDS.nwLat - BOUNDS.seLat)) * H;
  return { x, y };
}

// ---- Real Finnish places (true coords) -----------------------------------
const PLACES = [
  { name: 'Helsinki', lat: 60.170, lng: 24.938, kind: 'city' },
  { name: 'Porvoo', lat: 60.395, lng: 25.664, kind: 'town' },
  { name: 'Tampere', lat: 61.498, lng: 23.761, kind: 'city' },
  { name: 'Lahti', lat: 60.983, lng: 25.656, kind: 'town' },
  { name: 'Jyväskylä', lat: 62.242, lng: 25.747, kind: 'town' },
  { name: 'Kuopio', lat: 62.892, lng: 27.678, kind: 'town' },
  { name: 'Joensuu', lat: 62.601, lng: 29.763, kind: 'town' },
  { name: 'Koli NP', lat: 63.098, lng: 29.816, kind: 'park' },
];

// A few large lakes (stylized blobs at roughly true positions) for place-truth.
const LAKES = [
  { name: 'Saimaa', lat: 61.6, lng: 28.3, rx: 90, ry: 130 },
  { name: 'Päijänne', lat: 61.6, lng: 25.5, rx: 34, ry: 120 },
  { name: 'Pielinen', lat: 63.25, lng: 29.55, rx: 40, ry: 78 },
  { name: 'Näsijärvi', lat: 61.7, lng: 23.9, rx: 26, ry: 60 },
];

// Coastline / Gulf of Finland hint: everything south of a gentle curve is sea.
// We'll just tint the bottom-left (Gulf of Finland + Baltic) as water.

// ---- Build SVG ------------------------------------------------------------
const esc = (s) => String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;');

let svg = '';
svg += `<svg xmlns="http://www.w3.org/2000/svg" width="${W}" height="${H}" viewBox="0 0 ${W} ${H}">`;

// defs: paper noise + hatch + drop
svg += `<defs>
  <filter id="paper"><feTurbulence type="fractalNoise" baseFrequency="0.9" numOctaves="2" result="n"/><feColorMatrix in="n" type="saturate" values="0"/><feComponentTransfer><feFuncA type="linear" slope="0.05"/></feComponentTransfer><feComposite operator="over" in2="SourceGraphic"/></filter>
  <pattern id="hatch" width="8" height="8" patternTransform="rotate(45)" patternUnits="userSpaceOnUse"><line x1="0" y1="0" x2="0" y2="8" stroke="#c3d0c3" stroke-width="1.6"/></pattern>
  <pattern id="water" width="10" height="10" patternUnits="userSpaceOnUse"><rect width="10" height="10" fill="#b9ccd2"/><line x1="0" y1="5" x2="10" y2="5" stroke="#a6bfc7" stroke-width="1"/></pattern>
</defs>`;

// Land base (kraft paper)
svg += `<rect width="${W}" height="${H}" fill="#e4d9bf"/>`;
svg += `<rect width="${W}" height="${H}" fill="url(#hatch)" opacity="0.35"/>`;

// Gulf of Finland / Baltic (bottom strip, below Helsinki latitude ~60.1)
const sea = project(60.05, BOUNDS.nwLng).y;
svg += `<path d="M0,${H} L0,${sea} Q${W*0.35},${sea+40} ${W*0.62},${sea-10} T${W},${sea+30} L${W},${H} Z" fill="url(#water)" stroke="#7f9aa2" stroke-width="2"/>`;

// A stylized eastern border (Finland/Russia) — soft dashed line down the right.
const bTop = project(64.0, 30.4); const bBot = project(60.6, 30.75);
svg += `<path d="M${bTop.x},4 C${bTop.x-30},${H*0.3} ${bBot.x-20},${H*0.55} ${bBot.x},${bBot.y}" fill="none" stroke="#9a7d63" stroke-width="2.5" stroke-dasharray="2 8" opacity="0.7"/>`;

// Lakes
for (const lk of LAKES) {
  const p = project(lk.lat, lk.lng);
  svg += `<ellipse cx="${p.x.toFixed(1)}" cy="${p.y.toFixed(1)}" rx="${lk.rx}" ry="${lk.ry}" fill="url(#water)" stroke="#7f9aa2" stroke-width="1.5" transform="rotate(-18 ${p.x.toFixed(1)} ${p.y.toFixed(1)})"/>`;
}

// Trip route: Helsinki -> Porvoo -> ... -> Koli (dashed rust line)
const routeOrder = ['Helsinki','Porvoo','Lahti','Kuopio','Joensuu','Koli NP'];
const rpts = routeOrder.map(n => { const pl = PLACES.find(p=>p.name===n); const p=project(pl.lat,pl.lng); return `${p.x.toFixed(1)},${p.y.toFixed(1)}`; });
svg += `<polyline points="${rpts.join(' ')}" fill="none" stroke="#a4491f" stroke-width="3" stroke-dasharray="3 9" stroke-linecap="round" opacity="0.85"/>`;

// Graticule ticks (lat/lng labels — mono, field-guide)
for (let lat = 60; lat <= 64; lat++) {
  const p = project(lat, BOUNDS.nwLng);
  const ly = Math.max(p.y, 16);
  svg += `<line x1="0" y1="${p.y.toFixed(1)}" x2="14" y2="${p.y.toFixed(1)}" stroke="#6b5a41" stroke-width="1.5"/>`;
  svg += `<text x="18" y="${(ly+4).toFixed(1)}" font-family="Courier New,monospace" font-size="15" fill="#6b5a41">${lat}°N</text>`;
}
for (let lng = 24; lng <= 30; lng += 2) {
  const p = project(BOUNDS.seLat, lng);
  svg += `<line x1="${p.x.toFixed(1)}" y1="${H}" x2="${p.x.toFixed(1)}" y2="${H-14}" stroke="#6b5a41" stroke-width="1.5"/>`;
  svg += `<text x="${p.x.toFixed(1)}" y="${H-20}" font-family="Courier New,monospace" font-size="15" fill="#6b5a41" text-anchor="middle">${lng}°E</text>`;
}

// Places
for (const pl of PLACES) {
  const p = project(pl.lat, pl.lng);
  const isPark = pl.kind === 'park';
  const isCity = pl.kind === 'city';
  const r = isPark ? 11 : isCity ? 8 : 6;
  const fill = isPark ? '#3d5a3a' : isCity ? '#a4491f' : '#b9812e';
  svg += `<circle cx="${p.x.toFixed(1)}" cy="${p.y.toFixed(1)}" r="${r}" fill="${fill}" stroke="#2b2116" stroke-width="2"/>`;
  const dy = isPark ? -18 : 20;
  const weight = isPark ? '700' : '600';
  svg += `<text x="${p.x.toFixed(1)}" y="${(p.y+dy).toFixed(1)}" font-family="Georgia,serif" font-size="${isPark?20:16}" font-weight="${weight}" fill="#2b2116" text-anchor="middle" paint-order="stroke" stroke="#e4d9bf" stroke-width="3">${esc(pl.name)}</text>`;
}

// Title cartouche (field-guide)
svg += `<g transform="translate(28,28)">
  <rect x="0" y="0" width="316" height="74" fill="#f2ead8" stroke="#2b2116" stroke-width="2.5" rx="4"/>
  <text x="16" y="28" font-family="Courier New,monospace" font-size="12" letter-spacing="2" fill="#93805f">FIELD GUIDE · MAP PLATE I</text>
  <text x="16" y="56" font-family="Georgia,serif" font-size="26" font-weight="700" fill="#2b2116">Suomi · Finland</text>
</g>`;

// Compass rose (simple)
svg += `<g transform="translate(${W-70},70)" font-family="Georgia,serif">
  <circle r="34" fill="#f2ead8" stroke="#2b2116" stroke-width="2"/>
  <path d="M0,-30 L7,0 L0,30 L-7,0 Z" fill="#a4491f" stroke="#2b2116" stroke-width="1"/>
  <text y="-40" font-size="16" font-weight="700" text-anchor="middle" fill="#2b2116">N</text>
</g>`;

svg += `</svg>`;

const outDir = __dirname;
fs.writeFileSync(path.join(outDir, 'finland-map.svg'), svg);

// Emit the projection contract for the handoff + a self-check.
const check = PLACES.map(pl => { const p = project(pl.lat, pl.lng); return { name: pl.name, lat: pl.lat, lng: pl.lng, x: Math.round(p.x), y: Math.round(p.y) }; });
console.log(JSON.stringify({ BOUNDS, W, H, corners: {
  NW: project(BOUNDS.nwLat, BOUNDS.nwLng),
  SE: project(BOUNDS.seLat, BOUNDS.seLng),
  center: project((BOUNDS.nwLat+BOUNDS.seLat)/2, (BOUNDS.nwLng+BOUNDS.seLng)/2),
}, placePixels: check }, null, 2));
